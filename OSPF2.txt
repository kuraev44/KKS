1.настройка интерфейса
interface GigabitEthernet0/1.800
 encapsulation dot1Q 800
 ip address 192.168.1.126 255.255.255.128
2.настрйока DHCP
ip dhcp pool subnet1
 network 192.168.1.0 255.255.255.128
 default-router 192.168.1.126
ip dhcp pool subnet3
 network 192.168.1.160 255.255.255.240
 default-router 192.168.1.174

3.Настройка интерфейса Loopback 0 для идентификацции OSPF(под одному loopback на каждый интерфейс)
ip adress 192.168.10.2 255.255.255.255

4.Настройка процесса OSPF
router ospf 1
network 192.168.2.0 0.0.0.3 area 0
network 192.168.1.0 0.0.0.127 area 0
network 192.168.1.160 0.0.0.15 area 0

5.Настройка маршрута по умолчанию(на граничном роутере)
ip route 0.0.0.0 0.0.0.0 211.211.211.2

6.Включение режима обмена дефолтным маршрутом
router ospf 1
default-information originate

7.Объявление внутренних и внешних интерфейсов для сетевой трансляции
interface GigabitEthernet0/0
ip nat outside

interface GigabitEthernet0/0/0
 ip nat inside

 interface GigabitEthernet0/1/0
 ip nat inside

8.Создание access-list для NAT
permit 192.168.1.0 0.0.0.127
 permit 192.168.1.160 0.0.0.15
 permit 192.168.1.128 0.0.0.31

9.Команда включения трансляции NAT(overload)
ip nat inside source list NAT interface GigabitEthernet0/0 overload
